        // 將像素資料繪製到 Canvas
        function drawPixelArt(canvasId, pixelArray) {
            const canvas = document.getElementById(canvasId);
            if(!canvas) return;
            const ctx = canvas.getContext('2d');
            const size = 16; // 16x16 grid
            const pixelSize = canvas.width / size;

            ctx.clearRect(0, 0, canvas.width, canvas.height);

            for (let y = 0; y < size; y++) {
                if(!pixelArray[y]) continue;
                for (let x = 0; x < size; x++) {
                    const char = pixelArray[y][x];
                    let colorCode = '_';
                    
                    // 解析雙字元色碼 (如 P1, G2) 或單字元色碼
                    if (char !== '0' && char !== 'W' && char !== 'E' && char !== '_' && x < size - 1 && pixelArray[y][x+1].match(/[1-9A-Z]/)) {
                        colorCode = char + pixelArray[y][x+1];
                    } else if (char === 'P' || char === 'G' || char === 'B' || char === 'Y') {
                        // skip if it was part of a 2-char code read in previous loop
                        if (x > 0 && (pixelArray[y][x-1] === 'P' || pixelArray[y][x-1] === 'G' || pixelArray[y][x-1] === 'B' || pixelArray[y][x-1] === 'Y')) {
                            continue;
                        }
                    } else {
                        colorCode = char;
                    }

                    // 重新處理簡單字元對應
                    if(char === '0') colorCode = '0';
                    if(char === 'W') colorCode = 'W';
                    if(char === 'E') colorCode = 'E';
                    if(char === '_') colorCode = '_';

                    // 暴力解析 (簡化寫法)
                    let finalColor = 'transparent';
                    if(char === '_') finalColor = 'transparent';
                    else if(char === '0') finalColor = pal['0'];
                    else if(char === 'W') finalColor = pal['W'];
                    else if(char === 'E') finalColor = pal['E'];
                    else {
                         // 嘗試讀取兩碼
                         let twoChar = char + (pixelArray[y][x+1] || '');
                         if(pal[twoChar]) {
                             finalColor = pal[twoChar];
                             x++; // 跳過下一碼
                         } else if(pal[char]) {
                             finalColor = pal[char];
                         }
                    }

                    if (finalColor !== 'transparent') {
                        ctx.fillStyle = finalColor;
                        ctx.fillRect(x * pixelSize, y * pixelSize, pixelSize, pixelSize);
                    }
                }
            }
        }
